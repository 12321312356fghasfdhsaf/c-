#define _CRT_SECURE_NO_WARNINGS

#include <iostream>
#include <chrono>
#include <functional>

using namespace std;
int plus_func (int a, int b)
{
	return a + b;
}

int minus_func (int a, int b)
{
	return a - b;
}

int multi (int a, int b)
{
	return a * b;
}

int dev (int a, int b)
{
	return a / b;
}


int myF(int x, int y)
{
	return 3+ 4;
}

int main()
{
	int (*ptrT) (int, int) = &myF;

	//cout << (*ptrT)(3,4) << endl;


	std::function<int(int, int)> ptrToFunc = myF;

	ptrToFunc(7, 8);

	int x = 5, y=1;
	char choice;

	cin >> choice;

	switch (choice)
	{
	case '+':
	{
		ptrT = plus_func;
		break;
	}

	case '-':
	{
		ptrT = minus_func;
		break;
	}

	case '*':
	{
		ptrT = multi;
		break;
	}
	case '/':
	{
		ptrT = dev;
		break;
	}


	}

	cout<< (*ptrT)(x, y) << endl;


		
}
